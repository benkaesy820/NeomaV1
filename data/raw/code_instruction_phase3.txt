<instruction>
Create a Python helper that normalizes a title for storage.
</instruction>
<constraints>
- accept a string
- trim surrounding whitespace
- collapse internal whitespace to a single space
- raise ValueError if the result is empty
</constraints>
<answer>
import re


def normalize_title(title: str) -> str:
    normalized = re.sub(r"\s+", " ", title.strip())
    if not normalized:
        raise ValueError("title must not be empty")
    return normalized
</answer>

<instruction>
Write a Python function that safely reads a UTF-8 text file.
</instruction>
<constraints>
- accept pathlib.Path
- reject directories
- return the file text
- wrap decoding errors with a clear ValueError
</constraints>
<answer>
from pathlib import Path


def read_text_file(path: Path) -> str:
    if not path.exists():
        raise FileNotFoundError(f"file does not exist: {path}")
    if not path.is_file():
        raise ValueError(f"path is not a file: {path}")

    try:
        return path.read_text(encoding="utf-8")
    except UnicodeDecodeError as exc:
        raise ValueError(f"file is not valid UTF-8: {path}") from exc
</answer>

</instruction>

    for value in values:
        if value in allowed:
            result.append(value)
    return result

<answer>

<instruction>
Write a Python function that groups rows by a string field.
</instruction>
<constraints>
- accept list[dict]
- accept the field name
- use "unknown" when the field is missing or not a string
- return dict[str, list[dict]]
</constraints>
<answer>
def group_rows_by_field(rows: list[dict], field: str) -> dict[str, list[dict]]:
    groups: dict[str, list[dict]] = {}
    for row in rows:
        value = row.get(field)
        key = value if isinstance(value, str) and value else "unknown"
        groups.setdefault(key, []).append(row)
    return groups
</answer>

<instruction>
Create a TypeScript parser for a task object.
</instruction>
<constraints>
- input is unknown
- require id and title strings
- trim title
- return Result<Task>
</constraints>
<answer>
type Result<T> =
  | { ok: true; value: T }
  | { ok: false; error: string };

type Task = {
  id: string;
  title: string;
  completed: boolean;
};

function parseTask(input: unknown): Result<Task> {
  if (!input || typeof input !== "object") {
    return { ok: false, error: "task must be an object" };
  }

  const record = input as Record<string, unknown>;
  if (typeof record.id !== "string" || record.id.trim() === "") {
    return { ok: false, error: "task id is required" };
  }
  if (typeof record.title !== "string" || record.title.trim() === "") {
    return { ok: false, error: "task title is required" };
  }

  return {
    ok: true,
    value: {
      id: record.id.trim(),
      title: record.title.trim(),
      completed: record.completed === true,
    },
  };
}
</answer>

function indexById(items) {
  const indexed = {};
  for (const item of items) {
    if (item.id === undefined || item.id === null) {
      continue;
    }
    indexed[String(item.id)] = item;
  }
  return indexed;
}

<instruction>
Write a JavaScript function that retries an async operation.
</instruction>
<constraints>
- accept operation and maxAttempts
- retry only after failures
- throw the last error when all attempts fail
</constraints>
<answer>
async function retry(operation, maxAttempts) {
  let lastError;
  for (let attempt = 1; attempt <= maxAttempts; attempt += 1) {
    try {
      return await operation();
    } catch (error) {
      lastError = error;
    }
  }
  throw lastError;
}
</answer>

<instruction>
Write a PowerShell function that requires an existing directory.
</instruction>
<constraints>
- accept a path
- use LiteralPath
- throw when missing
- throw when it is not a directory
</constraints>
<answer>
function Require-Directory {
    param([string]$Path)

    if (-not (Test-Path -LiteralPath $Path)) {
        throw "Path does not exist: $Path"
    }

    $Item = Get-Item -LiteralPath $Path
    if (-not $Item.PSIsContainer) {
        throw "Path is not a directory: $Path"
    }

    return $Item
}
</answer>

Write a PowerShell function that returns file counts by extension.
</instruction>

- lowercase extensions
- use "<none>" when there is no extension
</constraints>
<answer>
function Get-ExtensionCounts {
    param([string]$Path)

    Get-ChildItem -LiteralPath $Path -Recurse -File |
        Group-Object {
            if ($_.Extension) { $_.Extension.ToLowerInvariant() } else { "<none>" }
        } |
        ForEach-Object {
            [PSCustomObject]@{
                Extension = $_.Name
                Count = $_.Count
            }
        }
}
</answer>

Write SQL that finds open tickets for one customer.
</instruction>

- join tickets to customers
- filter by customer id
- filter open status
- newest first
</constraints>
<answer>
SELECT
    tickets.id,
    tickets.title,
    tickets.created_at,
    customers.name AS customer_name
FROM tickets
JOIN customers ON customers.id = tickets.customer_id
WHERE customers.id = :customer_id
  AND tickets.status = 'open'
ORDER BY tickets.created_at DESC;

Improve this Python code to avoid repeated sorting inside a loop.
</instruction>

def top_names(groups):
    result = []
    for group in groups:
        result.append(sorted(group["names"])[0])
    return sorted(result)
</bad_code>
<reasoning>
The code sorts each full name list even though it only needs the smallest item.
Using min avoids unnecessary work.
</reasoning>
<answer>
def top_names(groups: list[dict]) -> list[str]:
    result: list[str] = []
    for group in groups:
        names = group.get("names", [])
        if names:
            result.append(min(names))
    return sorted(result)
</answer>

<instruction>
Write a Python function that redacts sensitive keys before logging.
</instruction>
<constraints>
- accept dict
- redact password, token, secret, api_key
- do not mutate the input
</constraints>
<answer>
SENSITIVE_KEYS = {"password", "token", "secret", "api_key"}


def redact_for_log(data: dict) -> dict:
    redacted: dict = {}
    for key, value in data.items():
        if key.lower() in SENSITIVE_KEYS:
            redacted[key] = "<redacted>"
        else:
            redacted[key] = value
    return redacted
</answer>

<instruction>
Write a TypeScript function that redacts sensitive fields before logging.
</instruction>
<constraints>
- accept Record<string, unknown>
- redact password, token, secret, apiKey
- return a new object
</constraints>
<answer>
function redactForLog(input: Record<string, unknown>): Record<string, unknown> {
  const sensitive = new Set(["password", "token", "secret", "apiKey"]);
  const output: Record<string, unknown> = {};

  for (const [key, value] of Object.entries(input)) {
    output[key] = sensitive.has(key) ? "<redacted>" : value;
  }

  return output;
}
</answer>
